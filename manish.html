EPISODE 1 – Real-Time AQI Prediction & Pollution Monitoring

What It Means

This system continuously measures air quality and predicts future AQI values using historical sensor data.

Why It Is Necessary

Air pollution causes respiratory diseases; AQI prediction allows preventive actions and alerts.

C++ Code

#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector aqi(n);

    for(int i = 0; i < n; i++)
        cin >> aqi[i];

    double sx=0, sy=0, sxy=0, sxx=0;
    for(int i = 0; i < n; i++) {
        sx += i;
        sy += aqi[i];
        sxy += i * aqi[i];
        sxx += i * i;
    }

    double b = (n*sxy - sx*sy) / (n*sxx - sx*sx);
    double a = (sy - b*sx) / n;

    int h;
    cin >> h;

    cout << "Predicted AQI: " << a + b*(n+h) << endl;
}

Sensors / Input

  • PMS5003, SDS011
  • MQ-135, MQ-7
  • ESP32 Data Upload
EPISODE 2 – Smart Industrial Emission Violation Detection

What It Means

Monitors chimney gas pollution & detects abnormal emissions.

Why Necessary

Prevents illegal night-time pollution & protects public health.

C++ Code

#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;

    vector e(n);
    for(int i = 0; i < n; i++) cin >> e[i];

    double mean = accumulate(e.begin(), e.end(), 0.0) / n;
    double var = 0;
    for(double x : e) var += (x-mean)*(x-mean);

    double sd = sqrt(var/n);

    for(int i = 0; i < n; i++)
        if(fabs(e[i] - mean) > 3 * sd)
            cout << "Violation at sensor " << i << endl;
}
EPISODE 3 – Urban Heat Island & Green Cover Optimization

C++ Code

#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;

    vector temp(n);
    for(int i = 0; i < n; i++) cin >> temp[i];

    vector cent(k);
    for(int i = 0; i < k; i++) cent[i] = temp[i];

    vector lbl(n);

    for(int it = 0; it < 20; it++) {
        for(int i = 0; i < n; i++) {
            double best = 1e9;
            for(int j = 0; j < k; j++)
                if(abs(temp[i]-cent[j]) < best) {
                    best = abs(temp[i]-cent[j]);
                    lbl[i] = j;
                }
        }

        vector sum(k), cnt(k);
        for(int i = 0; i < n; i++) {
            sum[lbl[i]] += temp[i];
            cnt[lbl[i]]++;
        }

        for(int j = 0; j < k; j++)
            cent[j] = sum[j]/cnt[j];
    }

    int hot = max_element(cent.begin(), cent.end()) - cent.begin();
    cout << "Hottest Cluster: " << hot << endl;
}

EPISODE 6 – Smart Noise Pollution Control

C++ Code

#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;

    vector s(n);
    for(int i = 0; i < n; i++) cin >> s[i];

    double sum = 0;
    for(double x : s) sum += x*x;

    double rms = sqrt(sum/n);

    if(rms > 0.6)
        cout << "Noise Violation\n";
    else
        cout << "Normal\n";
}

EPISODE 9 – Pollution-Aware Route for Schools & Hospitals

C++ Code

#include <bits/stdc++.h>
using namespace std;

int main() {
    int V, E;
    cin >> V >> E;

    vector>> g(V);
    for(int i = 0; i < E; i++) {
        int u,v,w;
        cin >> u >> v >> w;
        g[u].push_back({v,w});
        g[v].push_back({u,w});
    }

    int s, d;
    cin >> s >> d;

    vector dist(V, 1e9);
    priority_queue, vector>, greater<>> pq;

    dist[s] = 0;
    pq.push({0,s});

    while(!pq.empty()) {
        auto [cd,u] = pq.top(); pq.pop();
        for(auto &e : g[u]) {
            if(dist[e.first] > cd + e.second) {
                dist[e.first] = cd + e.second;
                pq.push({dist[e.first], e.first});
            }
        }
    }

    cout << "Min Pollution Path: " << dist[d] << endl;
}
